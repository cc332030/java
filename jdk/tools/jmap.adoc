
= jmap

导出 hprof

[source,shell]
----

pid=1

# 导出内存信息
jmap -dump:format=b,file=${pid}-1.hprof ${pid}

# 查看所有对象及数量，包括活跃以及非活跃的
jmap -histo ${pid} | less

# 查看活跃对象及数量
jmap -histo:live ${pid} | less
